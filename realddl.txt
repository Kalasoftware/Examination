CREATE TABLE  "COURSE_U" 
   (	"COURSE_ID" NUMBER, 
	"C_NAME" VARCHAR2(40), 
	 CONSTRAINT "PK_COUR" PRIMARY KEY ("COURSE_ID") ENABLE
   )


CREATE TABLE  "QUESTION_U" 
   (	"QID" NUMBER, 
	"QUETXT" VARCHAR2(4000), 
	"OP1" VARCHAR2(4000), 
	"OP2" VARCHAR2(4000), 
	"OP3" VARCHAR2(4000), 
	"OP4" VARCHAR2(4000), 
	"COPT" NUMBER, 
	"TESTN" NUMBER, 
	 CONSTRAINT "DUAL_PK" PRIMARY KEY ("QID", "QUETXT") ENABLE, 
	 CONSTRAINT "FK" FOREIGN KEY ("TESTN")
	  REFERENCES  "TEST_U" ("TESTN") ON DELETE CASCADE ENABLE
   )
// ON DELETE CASCADE ALAG SE LIKHA HU NAHI CAHLE TOH BATANA 
CREATE TABLE  "RESULT_U" 
   (	"RESULTID" NUMBER, 
	"SPID" NUMBER, 
	"EXAMDATE" VARCHAR2(255), 
	"SCORE" NUMBER, 
	"TESTN" NUMBER, 
	"USERID" NUMBER, 
	 CONSTRAINT "PK_RES" PRIMARY KEY ("RESULTID") ENABLE, 
	 CONSTRAINT "FK_RES" FOREIGN KEY ("SPID")
	  REFERENCES  "STUDENT_U" ("SPID") ON DELETE CASCADE ENABLE, 
	 CONSTRAINT "FK_TESTNU" FOREIGN KEY ("TESTN")
	  REFERENCES  "TEST_U" ("TESTN") ON DELETE CASCADE ENABLE, 
	 CONSTRAINT "FK_USER" FOREIGN KEY ("USERID")
	  REFERENCES  "EUSER" ("USERID") ON DELETE CASCADE ENABLE
   )

CREATE TABLE  "STUDENT_U" 
   (	"SPID" NUMBER, 
	"ROLLNO" NUMBER, 
	"USER_ID" NUMBER, 
	"NAME" VARCHAR2(50), 
	"COURSE_ID" NUMBER, 
	 CONSTRAINT "PK_STU" PRIMARY KEY ("SPID") ENABLE, 
	 CONSTRAINT "FK_ST" FOREIGN KEY ("USER_ID")
	  REFERENCES  "EUSER" ("USERID") ON DELETE CASCADE ENABLE, 
	 CONSTRAINT "FK_C" FOREIGN KEY ("COURSE_ID")
	  REFERENCES  "COURSE_U" ("COURSE_ID") ON DELETE CASCADE ENABLE
   )

CREATE TABLE  "TEST_U" 
   (	"TESTN" NUMBER, 
	"TESTNAME" VARCHAR2(30), 
	"TESTTIME" VARCHAR2(255), 
	"TESTFLAG" NUMBER, 
	"TESTSTART" NUMBER, 
	 CONSTRAINT "PK_TEST" PRIMARY KEY ("TESTN") ENABLE
   )
